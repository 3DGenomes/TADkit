  camera?: (number)[] | null;
  target?: (number)[] | null;
  fov?: number | null;
  near?: number | null;
  far?: number | null;
  translate?: number | null;
  fog?: boolean | null;
  fogNear?: number | null;
  fogFar?: number | null;
}

export class Lighting {
  ambient: string;
  color: string;
  intensity: number;
}

export class Axis {
  size: number;
  visible: boolean;
}

export class Navigation {
  autoRotate: boolean;
  autoRotateSpeed: number;
}

export class Widget3D extends Widget {
  frustum: Frustum; // spatial matrix for 3D view

  axis?: Axis | null;
  navigation?: Navigation | null;
  geometries?: Geometries | null;
}
